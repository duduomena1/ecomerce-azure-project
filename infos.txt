CREATE TABLE Products (
    id INT IDENTITY(1,1) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description NVARCHAR(MAX),
    price DECIMAL(18,2),
    image_url NVARCHAR(2083),
);

CREATE TABLE Users (
    id INT IDENTITY(1,1) PRIMARY KEY ,
    name NVARCHAR(255) NOT NULL,
    mail NVARCHAR(255) NOT NULL UNIQUE,
    phone NVARCHAR(20),
);
